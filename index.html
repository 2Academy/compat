const API_BASE = "https://api.npoint.io"; async function createBlob(data) { const res = await fetch(API_BASE, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(data) }); if (!res.ok) throw new Error("Не удалось создать комнату"); const json = await res.json(); return json.id; } async function getBlob(id) { const res = await fetch(`${API_BASE}/${id}`, { headers: { "Accept": "application/json" } }); if (!res.ok) return null; return await res.json(); } async function updateBlob(id, data) { const res = await fetch(`${API_BASE}/${id}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(data) }); if (!res.ok) throw new Error("Не удалось обновить"); return true; }
